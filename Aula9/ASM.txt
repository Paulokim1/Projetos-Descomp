STA @292 #Armazena o valor do acumulador em HEX4
STA @293 #Armazena o valor do acumulador em HEX5

STA @3 #Armazena o valor do acumulador em MEM[3]
STA @4 #Armazena o valor do acumulador em MEM[4]
STA @5 #Armazena o valor do acumulador em MEM[5]

STA @15 #Armazena o valor do acumulador em MEM[15] (LIMITE CENTENA DE MILHAR)
LDI $0 #Carrega o valor 0 para fazer comparações
STA @6 #Armazena o valor do acumulador em MEM[6]
LDI $1 #Carrega o valor 1 para fazer incrementos
STA @7 #Armazena o valor do acumulador em MEM[7]
LDI $10 #Carrega o valor 10 para fazer comparações de limite

STA @11 #Armazena o valor do acumulador em MEM[9]
MAIN_LOOP:
NOP # LOOP PRINCIPAL
LDA @355 #Carrega o acumulador com a leitura do botão KEY3
CEQ @7 #Compara o valor de KEY3 com o valor 1
JEQ @DEF_LIMITE #Jump para a linha da subrotina de definição de limite
LDA @354 #Carrega o acumulador com a leitura do botão KEY2
CEQ @7 #Compara o valor de KEY2 com o valor 1
JEQ @INCREMENTA #Jump para a linha da subrotina de incremento
JMP @MAIN_LOOP
DEF_LIMITE:
NOP # DEF_LIMITE
LDA @7 #Carrega o valor 1
STA @258 #Armazena o valor 1 no LEDR9
JMP @DEF_LIMITE
INCREMENTA:
NOP # INCREMENTA
LDA @7 #Carrega o valor 1
STA @257 #Armazena o valor 1 no LEDR8
JMP @INCREMENTA
